        // Logica di funzionamento dell'app

        function checkLogin() {

            const pinInput = document.getElementById('pin').value;

            

            // Puoi cambiare '1234' con il PIN reale che preferisci usare

            if (pinInput === '1234' || pinInput !== '') { 

                document.getElementById('login-screen').classList.add('hidden');

                document.getElementById('app-screen').classList.remove('hidden');

            } else {

                alert('PIN errato. Riprova.');

            }

        }


        function inviaOrdine() {

            const tavolo = document.getElementById('tavolo').value;

            const ordine = document.getElementById('ordine').value;

            const msgDiv = document.getElementById('success-msg');


            if (!tavolo || !ordine) {

                alert('Per favore, seleziona un tavolo e inserisci i piatti.');

                return;

            }


            // Qui l'ordine viene elaborato. 

            console.log("Ordine inviato per:", tavolo, "Contenuto:", ordine);


            // Messaggio di conferma a schermo

            msgDiv.innerText = "Comanda inviata con successo!";

            

            // Pulisce i campi dopo l'invio

            document.getElementById('ordine').value = '';

            document.getElementById('tavolo').value = '';


            setTimeout(() => {

                msgDiv.innerText = '';

            }, 3000);

        }
